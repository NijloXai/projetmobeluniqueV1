```typescript
import { anonClient } from './supabase-admin'

export async function loginAsAdmin(): Promise<{ accessToken: string; cookieHeader: string }> {
  const { data, error } = await anonClient.auth.signInWithPassword({
    email: process.env.TEST_ADMIN_EMAIL!,
    password: process.env.TEST_ADMIN_PASSWORD!,
  })
  if (error || !data.session) {
    throw new Error(`Auth failed: ${error?.message ?? 'no session'}`)
  }

  const accessToken = data.session.access_token
  const refreshToken = data.session.refresh_token

  // Format cookies attendu par @supabase/ssr createServerClient
  // Le cookie name est sb-<project-ref>-auth-token par defaut
  // Pour l'instance locale, le project ref est "default" dans config.toml
  const cookieValue = JSON.stringify({
    access_token: accessToken,
    refresh_token: refreshToken,
    token_type: 'bearer',
    expires_in: 3600,
    expires_at: Math.floor(Date.now() / 1000) + 3600,
  })
  // @supabase/ssr utilise un cookie chunke : sb-<ref>-auth-token.0, sb-<ref>-auth-token.1, etc.
  // Pour les tests, un seul cookie base64-encoded suffit si le chunk est < 3180 chars
  const cookieName = 'sb-127-auth-token'
  const encoded = Buffer.from(cookieValue).toString('base64')
  const cookieHeader = `${cookieName}=${encoded}`

  return { accessToken, cookieHeader }
}
```

NOTE: Le format exact du cookie depend de la config @supabase/ssr. Si le pattern `sb-127-auth-token` ne fonctionne pas, il faudra ajuster en fonction du project_id dans supabase/config.toml (generalement `project_id = "default"` => `sb-default-auth-token`). Tester empiriquement lors de la premiere execution.

 * Cree le user admin pour les tests (idempotent).
 * Utilise adminClient.auth.admin.createUser() au lieu de SQL raw (recommandation RESEARCH.md).

export async function seedAdminUser(): Promise<string> {
  const { data: existing } = await adminClient.auth.admin.listUsers()
  const found = existing?.users?.find(u => u.email === TEST_ADMIN_EMAIL)
  if (found) return found.id

  const { data, error } = await adminClient.auth.admin.createUser({
    email: TEST_ADMIN_EMAIL,
    password: TEST_ADMIN_PASSWORD,
    email_confirm: true,
  })
  if (error) throw new Error(`seedAdminUser failed: ${error.message}`)
  return data.user.id
}
