const [modelsResult, fabricsResult, visualsResult] = await Promise.all([
  supabase
    .from('models')
    .select('*, model_images(*)')
    .eq('is_active', true)
    .order('created_at', { ascending: false }),

    .from('generated_visuals')
    .select('*, fabric:fabrics(*), model_image:model_images(*)')

    .eq('is_published', true)

])
